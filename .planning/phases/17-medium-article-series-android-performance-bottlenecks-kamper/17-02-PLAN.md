```kotlin
private fun setupKamper() {
    lifecycle.addObserver(Kamper)
    Kamper.apply {
        install(CpuModule)
        install(FpsModule)
        install(MemoryModule(applicationContext))
        install(NetworkModule)
        install(JankModule(application))
        install(GcModule)
        install(ThermalModule(applicationContext))
        install(
            IssuesModule(
                context = applicationContext,
                anr = AnrConfig(thresholdMs = 5_000L),
                slowStart = SlowStartConfig(coldStartThresholdMs = 1_500L),
                strictMode = StrictModeConfig()
            ) {
                crash { chainToPreviousHandler = false }
            }
        )

        addInfoListener<CpuInfo>     { if (it != CpuInfo.INVALID)     cpuScreen.update(it) }
        addInfoListener<FpsInfo>     { if (it != FpsInfo.INVALID)     fpsScreen.update(it) }
        addInfoListener<MemoryInfo>  { if (it != MemoryInfo.INVALID)  memoryScreen.update(it) }
        addInfoListener<NetworkInfo> { if (it != NetworkInfo.INVALID) networkScreen.update(it) }
        addInfoListener<IssueInfo>   { issuesScreen.addIssue(it.issue) }
        addInfoListener<JankInfo>    { if (it != JankInfo.INVALID)    jankScreen.update(it) }
        addInfoListener<GcInfo>      { if (it != GcInfo.INVALID)      gcScreen.update(it) }
        addInfoListener<ThermalInfo> { if (it != ThermalInfo.INVALID) thermalScreen.update(it) }
    }
}
```

    class MainActivity : AppCompatActivity() {

        private val cpuScreen     = CpuScreen()
        private val fpsScreen     = FpsScreen()
        private val memoryScreen  = MemoryScreen()
        private val networkScreen = NetworkScreen()
        private val issuesScreen  = IssuesScreen()
        private val jankScreen    = JankScreen()
        private val gcScreen      = GcScreen()
        private val thermalScreen = ThermalScreen()

        private val screens = listOf(
            cpuScreen, fpsScreen, memoryScreen, networkScreen,
            issuesScreen, jankScreen, gcScreen, thermalScreen
        )
        private val tabTitles = listOf("CPU", "FPS", "Memory", "Network", "Issues", "Jank", "GC", "Thermal")

        override fun onCreate(savedInstanceState: Bundle?) {
            super.onCreate(savedInstanceState)
            setContentView(R.layout.activity_main)
            setupTabs()
            setupKamper()
        }

        private fun setupTabs() {
            val tabLayout = findViewById<TabLayout>(R.id.tabLayout)
            tabTitles.forEach { tabLayout.addTab(tabLayout.newTab().setText(it)) }

            val fm = supportFragmentManager
            fm.beginTransaction().apply {
                screens.forEachIndexed { i, frag ->
                    add(R.id.fragmentContainer, frag, "tab_$i")
                    if (i != 0) hide(frag)
                }
            }.commit()

            tabLayout.addOnTabSelectedListener(object : TabLayout.OnTabSelectedListener {
                override fun onTabSelected(tab: TabLayout.Tab) {
                    supportFragmentManager.beginTransaction().apply {
                        screens.forEachIndexed { i, frag ->
                            if (i == tab.position) show(frag) else hide(frag)
                        }
                    }.commit()
                }
                override fun onTabUnselected(tab: TabLayout.Tab) {}
                override fun onTabReselected(tab: TabLayout.Tab) {}
            })
        }

        // Central install (D-15) — all 8 Kamper modules installed once here.
        // Per-screen Fragments do NOT install or uninstall Kamper. They only define
        // a public update(info) method that this Activity's listeners route to.
        // chainToPreviousHandler = false is DEMO-ONLY; production apps must set true.
        private fun setupKamper() {
            lifecycle.addObserver(Kamper)
            Kamper.apply {
                install(CpuModule)
                install(FpsModule)
                install(MemoryModule(applicationContext))
                install(NetworkModule)
                install(JankModule(application))
                install(GcModule)
                install(ThermalModule(applicationContext))
                install(
                    IssuesModule(
                        context = applicationContext,
                        anr = AnrConfig(thresholdMs = 5_000L, ignoreWhenDebuggerAttached = true),
                        slowStart = SlowStartConfig(coldStartThresholdMs = 1_500L),
                        strictMode = StrictModeConfig()
                    ) {
                        crash { chainToPreviousHandler = false } // demo only — production: true
                        memoryPressure {}
                        droppedFrames {}
                        slowSpan {}
                    }
                )

                addInfoListener<CpuInfo>     { if (it != CpuInfo.INVALID)     cpuScreen.update(it) }
                addInfoListener<FpsInfo>     { if (it != FpsInfo.INVALID)     fpsScreen.update(it) }
                addInfoListener<MemoryInfo>  { if (it != MemoryInfo.INVALID)  memoryScreen.update(it) }
                addInfoListener<NetworkInfo> { if (it != NetworkInfo.INVALID) networkScreen.update(it) }
                addInfoListener<IssueInfo>   { issuesScreen.addIssue(it.issue) }
                addInfoListener<JankInfo>    { if (it != JankInfo.INVALID)    jankScreen.update(it) }
                addInfoListener<GcInfo>      { if (it != GcInfo.INVALID)      gcScreen.update(it) }
                addInfoListener<ThermalInfo> { if (it != ThermalInfo.INVALID) thermalScreen.update(it) }
            }
        }
    }
